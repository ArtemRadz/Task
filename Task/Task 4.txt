#include <QApplication>
#include <QRadioButton>
#include <QVBoxLayout>
#include <QWidget>
#include <QPushButton>
#include <Windows.h>

int main(int argc, char *argv[])
{
    QApplication app(argc, argv);
    QRadioButton *rButtonRed = new QRadioButton("Red");
    QRadioButton *rButtonOrange = new QRadioButton("Orange");
    QRadioButton *rButtonGreen = new QRadioButton("Green");
    QVBoxLayout *hLayout = new QVBoxLayout;
    QPushButton *buttonStart = new QPushButton("Start");
    QPushButton *buttonStop = new QPushButton("Stop");
    QWidget *window = new QWidget;
    QObject::connect(buttonStart,SIGNAL(clicked()),rButtonRed,SLOT(click()));
    window->setWindowTitle("Task 3");
    window->setFixedSize(500,300);
    hLayout->addWidget(rButtonRed);
    hLayout->addWidget(rButtonOrange);
    hLayout->addWidget(rButtonGreen);
    hLayout->addWidget(buttonStart);
    hLayout->addWidget(buttonStop);
    window->setLayout(hLayout);
    window->show();
    return app.exec();
}
